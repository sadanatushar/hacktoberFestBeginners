#include <stdio.h>

int main(void) {
	int t;
	scanf("%i",&t);
	for(int z=0;z<t;z++)
	{
	    int n,m,k,l,r;
	    scanf("%d%d%d%d%d",&n,&m,&k,&l,&r);
	    int c[n],p[n];
	    for(int i=0;i<n;i++)
	        scanf("%d%d",&c[i],&p[i]);
	    for(int i=0;i<n;i++)
	    {
	        for(int j=0;j<n-i-1;j++)
	        {
	            if(p[j]>p[j+1])
	            {
	                int temp;
	                temp=p[j+1];
	                p[j+1]=p[j];
	                p[j]=temp;
	                temp=c[j+1];
	                c[j+1]=c[j];
	                c[j]=temp;
	            }
	        }
	    }
	    int key=-1;
	    for(int i=0;i<n;i++)
	    {
	        int time=m;
	        while(time>0)
	        {
	            if(c[i]>k+1)
	                c[i]-=1;
	            else if(c[i]<k-1)
	                c[i]+=1;
	            else
	                c[i]=k;
	            time--;
	        }
	        if(c[i]>=l&&c[i]<=r)
	        {
	            key=p[i];
	            break;
	        }
	    }
	    printf("%d\n",key);
	}
	return 0;
}
